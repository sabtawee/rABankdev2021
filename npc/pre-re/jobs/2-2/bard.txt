	mes " ท่านคงต้องการจะเปลี่ยนแค่อาชีพอย่างเดียว?";	
	next;
	menu " ใช่",L_CHANGE," รอข้าก่อน",-;

	mes "[Lalo]";
	mes " โอเค ข้าจะรอเจ้าที่นี่";
	close;

L_CHANGE:
	if(skillpoint != 0) goto L_SKILLSLEFT;
	callfunc "Job_Change",Job_Bard;
	callfunc "F_ClearJobVar";

	mes "[Lalo]";
	if (@gift==0) goto L_NOGIFT;

	mes " ดีมาก.. ข้าจะสร้างมันขึ้นมาให้ท่าน";
	mes " เป็นของที่ระลึกสำหรับการเปลี่ยนอาชีพ";
	mes " รอสักครู่";
	next;

	switch(@gift){
		case 1:
			delitem 1066,60;	//Fine-grained_Trunk + 50 JobLvl --> Harp [2]
			getitem 1910,1;
			break;
		case 2:
			delitem 1066,60;	//Fine-grained_Trunk --> Lute [2]
			getitem 1905,1;
			break;
		case 3:
			delitem 1067,60;	//Solid_Trunk --> Mandolin [2]
			getitem 1903,1;
			break;
		case 4:
			delitem 1068,60;	//Barren_Trunk --> Mandolin [2]
			getitem 1903,1;
			break;
		case 5:
			delitem 1019,60;	//Trunk --> Violin [3]
			getitem 1901,1;
	}

	emotion ET_CONGRATULATION;
	mes "[Lalo]";
	mes " นี่คือของขวัญของท่าน เมื่อท่านร้องเพลงมัน จะมีเสียงคลอไป";
	mes " หวังว่าท่านจะร้องเพลงที่มอบความสุข";
L_NOGIFT:
	mes " หวังว่าท่านจะร้องเพลงที่มอบความสุข";
	next;
L_CLOSE:
	mes "[Lalo]";
	mes " ไว้เจอกันคราวหน้า!";
	close;
}
