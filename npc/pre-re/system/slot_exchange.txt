morocc,148,83,6	script	แลกแต้ม Charm	112,{
	
L_Main:
	
	if($@slotEX_chk){
		end;
	}else{
		mes .npcname$;
		mes " ขณะนี้คุณมีแต้มวิจัย : ^0800ff" + #EXCHANGEPOINTS + "^000000 พ้อยท์";
		mes " โปรดเลือกของที่ต้องการแลกด้านล่าง";
				
		for(.@i=0;.@i<getarraysize(.exchange);set .@i,.@i+3){
			if(countitem(.exchange[.@i]) && getequipid(.exchange[.@i+1]) != .exchange[.@i]) {
				set .@menu$, .@menu$ + "  ^031cfc"+ getitemname(.exchange[.@i]) + "^000000 [^ff0000" + .exchange[.@i+2] + "^000000 พ้อยท์]:";
			}else{
				set .@menu$, .@menu$ + ":";
			}
		}
		set .@sel, select(.@menu$);
		.@sel = ((.@sel-1)*3);
		deletearray .@menu$[0],getarraysize(.@menu$);
		next;
		mes .npcname$;
		mes "-------------------------";
		mes " ไอเท็ม : ^031cfc" + getitemname(.exchange[.@sel]) + "^000000";
		mes " พ้อยท์ : ^ff0000" + .exchange[.@sel+2] + "^000000 พ้อยท์";
		
		menu " ~ ตกลง",-," ~ ย้อนกลับ",-,"  ยกเลิก",-;
		set .@menu2,@menu;
		next;
		if(.@menu2 == 1){
			mes .npcname$;
			mes "-------------------------";
			mes "ทำการแลกเปลี่ยนเสร็จสิ้น";
			mes "-------------------------";
			mes "ได้รับแต้มวิจัย : ^ff0000" + .exchange[.@sel+2] + "^000000 พ้อยท์";
			close2;
			delitem .exchange[.@sel],1;
			#EXCHANGEPOINTS += .exchange[.@sel+2];
			end;
		}
		if(.@menu2 == 2){
			goto L_Main;
		}
		end;
		
	}
	end;
	
OnInit:
	.npcname$ = " ^0004ff:[ แลกแต้มวิจัย ]: ^000000";
	setunitdata getnpcid(0),UNPC_GROUP_ID,2;
	waitingroom "ส่วมใส่ [] แลกแต้ม",0;
	
	////////////////////////////////////////////////////////////
	// กำหนดของ
	// ตัวอย่าง
	// <item id>,<Slot ที่อุปกรณ์นั้นใส่ได้ เอาไว้ตรวจสอบ>,< พ้อยท์ที่จะได้>{n,n};
	
	setarray .exchange[0],2331,EQI_ARMOR,2,2504,EQI_GARMENT,3,2404,EQI_SHOES,2,2406,EQI_SHOES,3,2402,EQI_SHOES,1,2310,EQI_ARMOR,1,2322,EQI_ARMOR,2,2313,EQI_ARMOR,1,2326,EQI_ARMOR,2,2502,EQI_GARMENT,1,2102,EQI_HAND_L,3,2104,EQI_HAND_L,2,2106,EQI_HAND_L,2;
	
	if((getarraysize(.exchange)%3) != 0){
		debugmes "\033[0;31mSlot Exchange setting is invalid, NPC disabled";
		$@slotEX_chk = 1;
	}
	end;	
	
}
